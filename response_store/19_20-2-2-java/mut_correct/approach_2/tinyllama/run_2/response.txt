public class IntegeRStack {
    public void push(int num) {
        stack.push(num);
    }
    public int pop() {
        if (stack.isEmpty()) {
            throw new EmptyStackException();
        }
        int result = stack.pop();
        return result;
    }
    public int peek() {
        return stack.peek();
    }
    public void push(int num, int op) {
        if (op == 0) { // Operator is zero
            stack.push(num);
        } else if (op == 1) { // Operator is minus (-)
            if (!isNegativeOperand(stack.peek())) { // No negative numbers to push
                throw new InvalidInputException("Invalid input: Negative operands cannot be pushed.");
            }
            stack.push(Math.abs(stack.pop()));
        } else if (op == 2) { // Operator is times (*)
            if (!isNumericalOperand(num)) {
                throw new InvalidInputException("Invalid input: Numeric operands cannot be pushed.");
            }
            stack.push(Math.multiplyExact(stack.pop(), num));
        } else if (op == 3) { // Operator is division (/)
            int top = peek();
            if (!isNumericalOperand(top)) {
                throw new InvalidInputException("Invalid input: Numeric operands cannot be pushed.");
            }
            stack.push(Math.divideExact(stack.pop(), top));
        } else { // Invalid operator (e.g. No valid minus, times or division)
            throw new InvalidInputException("Invalid operator: " + op);
        }
    }
    public int pop() {
        if (stack.isEmpty()) {
            throw new EmptyStackException();
        }
        int result = stack.pop();
        return result;
    }
    private boolean isNegativeOperand(int num) {
        // Check for negative numbers
        return num < 0 && (num == 0 || Math.signum(num) != Math.signum(stack.peek()));
    }
    private boolean isNumericalOperand(int num) {
        return !isNegativeOperand(num); // Check for negative operands
    }
}
