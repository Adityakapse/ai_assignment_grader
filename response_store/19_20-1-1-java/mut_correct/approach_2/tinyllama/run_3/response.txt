```cpp
#include <iostream>
#include <algorithm> // for find_if()

using namespace std;

int gcd(int a, int b) {
    if (a < b) {
        int tmp = a;
        a = b;
        b = tmp;
    }
    int r = a % b;
    if (r == 0) {
        return b;
    }
    else {
        return gcd(r, b);
    }
}

int lcm(int a, int b) {
    return (a * b) / gcd(a, b);
}

void findLCM(vector<int> &arr, int start, int end, int lcm, int &gcd, int &lcmIndex) {
    if (end <= start) { // base case: no common factors between two ranges
        cout << "No LCM found.";
        return;
    }
    
    int mid = start + (end - start) / 2; // find midpoint using binary search
    int gcdFound = gcd(arr[start], arr[mid]); // recursive call to find LCM of two ranges
    
    if (gcdFound == lcm) { // found LCM, increment lcmIndex and return
        cout << "LCM: " << gcdFound;
        lcmIndex++;
        return;
    } else if (gcdFound < lcm) { // start new search with smaller range
        end = mid - 1;
    } else { // end new search with larger range
        start = mid + 1;
    }
    
    findLCM(arr, start, end, lcm, gcd, lcmIndex);
}

int main() {
    int arr[] = {10, 20, 30, 40}; // input array of integers
    int n = sizeof(arr) / sizeof(arr[0]); // number of elements in array
    
    int lcm;
    
    cout << "Enter the first range of values (separated by commas): ";
    cin >> arr[0];
    
    findLCM(arr, 0, n-1, lcm, gcd, 0); // call function with start=0 and end=n-1
    
    cout << "LCM: " << lcm;
    
    return 0;
}
```

This implementation uses recursive function calls to find the LCM of two ranges. It starts by finding the largest common divisor (LCD) using recursion, then calls the `findLCM` function with smaller and larger ranges as input values, respectively. The output is printed at the end.
