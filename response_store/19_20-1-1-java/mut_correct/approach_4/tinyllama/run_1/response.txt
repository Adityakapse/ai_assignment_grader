import math # necessary for gcd() and lcm() functions

def gcd_impl(a, b):
    """
    Implementation of GCD using Euclid's Algorithm.
    :param a: first integer to find its GCD with b.
    :param b: second integer to find its GCD with a.
    :returns: the greatest common divisor (GCD) of a and b, rounded to two decimal places. If either input is 0 or None, returns None.
    """
    if a != 0 or b != 0:
        gcd = a % b
        while gcd != 0:
            b = gcd
            gcd = a % b
        return gcd
    else:
        return None

def lcm_impl(a, b):
    """
    Implementation of LCM using Euclid's Algorithm.
    :param a: first integer to find its LCM with b.
    :param b: second integer to find its LCM with a.
    :returns: the least common multiple (LCM) of a and b, rounded to two decimal places. If either input is 0 or None, returns None.
    """
    if a == 0 and b != 0:
        return None
    else:
        lcm = (a * b) // gcd_impl(a, b)
