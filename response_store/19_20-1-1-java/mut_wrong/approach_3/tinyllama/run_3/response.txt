PROBLEM STATEMENT:
Write a C++ program that prompts the user to input two positive integers n and k, and then calculates and outputs the largest prime factor(s) of the product of n and k. The program should include error handling for invalid inputs (e.g., non-integer numbers, negative integers, and/or zero divisors). The output should display the largest factor(s) in a clear and easy to read format (e.g., no need for complex mathematics terminology). Use appropriate variable names and comments where necessary.

SOLUTION:
```c++
#include <iostream>
#include <iomanip>
using namespace std;

int largest_prime(int n, int k) {
    int prime_factors[100] = { 1 };
    for (int I = 2; I * I <= n && I * I <= k; i++) {
        if (n % i == 0) {
            for (int j = i; j * j <= n; j++) prime_factors[j]++;
        }
    }
    
    int max_factor = 1;
    for (int I = 2; I < 100 && max_factor * I <= k; i++) {
        if (prime_factors[i] > max_factor) {
            max_factor = prime_factors[i];
        }
    }
    
    return max_factor;
}

int main() {
    int n, k;
    while (cin >> n >> k && n <= 0 || k <= 0) {
        if (!(cin >> n >> k)) break;
        
        cout << "Product of " << n << " and " << k << " is: " << largest_prime(n, k) << endl;
    }
    
    return 0;
}
```
