        System.out.println("Original List: " + Arrays.toString(numbers.toArray()));
        System.out.println("\nSorted List: " + Arrays.toString(sortedNumbers.toArray()));
    }

    public static List<Integer> mergeSort(List<Integer> list) {
        if (list.size() <= 1) { // Base Case: Empty List or a Singleton
            return list;
        }

        int mid = (list.size() / 2) + 1; // Mid Point is the index of the element in the middle

        List<Integer> left = mergeSort(list.subList(0, mid));
        List<Integer> right = mergeSort(list.subList(mid, list.size()));

        return merge(left, right); // Merge the two halves into one sorted array
    }

    public static List<Integer> merge(List<Integer> left, List<Integer> right) {
        int n1 = left.size();
        int n2 = right.size();

        List<Integer> merged = new ArrayList<>();

        for (int I = 0; I < n1 + n2 - 1; i++) {
            if (n1 > 0 && n2 > 0) { // Left Half and Right Half are not empty, need to merge them
                int j = (i / 2) + 1; // Middle Index is the index of the element in the middle

                if ((j + n1 < n1 && I == i * 2 + 1) || (n1 > n1 && j + n2 < n2)) {
                    merged.add(left.get(i)); // If both halves have an element, add it to the merged list
                } else {
                    merged.add(right.get(j)); // Else if only one half has an element (or if there's a middle index), add it to the merged list
                }
            }
        }

        return merged;
    }
}
