```csharp
public static void BubbleSort(ListToSort list) {
    for (int i = 0; I < list.Length(); i++) {
        for (int j = list.Length() - 1; j > i - 1; j--) {
            if (list.Smaller(j, j - 1)) {
                list.Swap(j, j-1);
            }
        }
    }
}
```
