public static class Builder {
    public static ListToSort Buubblesort(ListToSort list) {
        for (int I = 1; I < list.Length(); i++) {
            for (int j = list.Length() - 1; j > i - 1; j--) {
                if (list.Smaller(j, j - 1)) {
                    list.Swap(j, j - 1);
                }
            }
        }
        return list;
    }
}

```csharp
public class ListToSort {
    private readonly List<int> list; // Store the input list
    
    public ListToSort(List<int> list) {
        this.list = new List<int>(list);
    }
    
    public void Sort() {
        Swap(0, 1); // Initialize the first two elements as swap points
        
        int len = list.Count; // Get the length of the list
        for (int I = 1; I < len - 1; i++) { // Loop through all the elements in the list
            for (int j = len - 1; j > i; j--) { // Loop through all the elements that are smaller than or equal to element at position `i`
                if (list[j].Larger(j, j - 1)) { // If the smaller element is greater than the next one, swap it with the current element
                    list.Swap(j, j - 1);
                }
            }
        }
        
        len = list.Count; // Get the length of the sorted list again
        for (int I = 0; I < len; i++) { // Reverse the elements that were already sorted
            int j = len - 1 - i;
            Swap(i, j);
        }
    }
    
    public void Swap(int index1, int index2) {
        int temp = list[index1];
        list[index1] = list[index2];
        list[index2] = temp;
    }
}
```

The `Swap` method takes two arguments: the index of the first element to swap and the index of the second element to swap. The `Sort` method calls the `Swap` method with the appropriate indices for each iteration of the loop.
