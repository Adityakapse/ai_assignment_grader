```csharp
        return false;
    }
    
    // Swap the key with the middle element in the heap
    int middle = heap.Length() / 2;
    heap.Swap(middle, heap.IndexOf(key));
    
    if (heap.Smaller(key, heap.GetMin())) {
        // Heap[min] is less than or equal to key, so heap contains a smaller element and its key matches key
        return true;
```

```csharp
public static void Heapify(Heap heap) {
```

```csharp
    public static ListToSort Bubblesort(ListToSort list) {
        for (int I = 1; I < list.Length(); i++) {
            for (int j = list.Length() - 1; j > I - 1; j--) {
                if (list.Smaller(j, j - 1)) {
                    list.Swap(j, j - 1);
                }
            }
        }
        return list;
    }
    
    public static void Swap(int x, int y) {
        int temp = list[x];
        list[x] = list[y];
        list[y] = temp;
    }
    
    public static int Length() => System.Math.Min(list.Count + 1, list.Max() - list.Min()) + 1;
}
```

This implementation uses the `Length()` method to bound all relevant loops, including the inner loop and the outer loop. The algorithm uses the `Smaller()` method to ensure that elements are swapped in ascending order.
