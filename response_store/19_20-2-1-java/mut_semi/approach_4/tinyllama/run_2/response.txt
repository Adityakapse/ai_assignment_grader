public class Queue {
    private int start; // Start index of the buffer array
    private int end; // End index of the buffer array
    private String[] buffer; // Array of String elements stored in the buffer array
    
    public boolean isEmpty() { return start == end; } // Check if the queue is empty
    public String head() { return buffer[start]; } // Return the front element, or null if queue is empty
    public void enqueue(String s) { buffer[end++] = s; } // Insert the specified element at the end of the buffer array
    
    public String dequeue() {
        String s = buffer[start]; // Get and return the front element, or null if queue is empty
        start++; // Increment the index of the front element
        if (start == 100) { // Check if end index has already reached 100, in case queue was enqueued with a length greater than 100
            start = 0; // If so, reset start index to 0 and continue with the previous code
        }
        return s; // Return the front element or null if it's not available
    }
    
    public String[] getBuffer() { // Return an array of the elements stored in the buffer array
        return buffer;
    }
}
