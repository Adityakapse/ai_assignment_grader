    def emptyqueue(self):
        return all(i in self.buffer for i in range(len(self.buffer) - 1))
    
    def head(self):
        return list(filter(None, self.buffer)[0]) if not self.emptyqueue() else None
    
    def enqueue(self, s):
        self.buffer[len(self.buffer) // 2] = s
        self.enqueuepop()
    
    def dequeue(self):
