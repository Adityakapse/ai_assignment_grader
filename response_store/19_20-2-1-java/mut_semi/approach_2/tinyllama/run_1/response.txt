```python
class Queue:
     def __init__(self):
         self.head = None
         self.tail = None
         self._buffer = []  # stores elements in a list
         
     def isempty(self):
         return self.head == None or self.tail == None

     def enqueue(self, item):
         if self.isempty():
             self.head = item
             self.tail = item
         else:
             self.tail = self.tail.next  # update tail pointer
             self._buffer.append(item)  # add item to front of buffer
         
     def dequeue(self):
         if self.isempty():
             raise ValueError("Queue is empty")
         temp = self.head
         self.head = self.head.next  # update head pointer
         self.tail = self.tail.prev  # update tail pointer
         return temp
```
