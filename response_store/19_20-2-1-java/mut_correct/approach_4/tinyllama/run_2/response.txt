Sample Solution:

Public Class Queue {
    private int start = 0;
    private int end = 0;
    private String[] buffer = new String[100]; // internal storage and tracking state
    
    public boolean isEmptyQueue() { return start == end; } // check if queue has been emptied completely
    public void enqueue(String s) { // insert element in front of the queue
        buffer[end] = s; // store new element at end of buffer
        end++; // increment end index
        if (end == 100) end = 0; // stop if buffer is full
    }
    
    public String dequeue() { // remove and return first element in queue
        String s = buffer[start]; // read current element at start of buffer
        start++; // increment start index
        if (start == 100) start = 0; // stop if buffer is full
        return s; // return retrieved element
    }
}
