class Queue:

        self.end = 0
        self.buffer = []

    def emptyqueue(self):
        return self.start == self.end

    def head(self):
        return self.buffer[self.start]

    def enqueue(self, s):
        self.buffer[self.end] = s
        self.end += 1
        if self.end == len(self.buffer): self.end = 0
    
    def dequeue(self):
        s = self.buffer[self.start]
        self.buffer[self.end] = None
        self.start += 1
        return s
