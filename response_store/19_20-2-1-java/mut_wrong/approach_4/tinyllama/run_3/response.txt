public class Stack {
     private int size; // Size of stack (maximum capacity)
     private Node head; // Head pointer to first element in the stack
     private Node tail; // Tail pointer to last element in the stack
     
     // Constructor
     public Stack() {
         head = null;
         tail = null;
         size = 0;
     }
     
     public int sizeOfStack() {
         return size;
     }
     
     // Push element onto the stack
     public void push(int x) {
         Node newNode = new Node(x); // Create a new node to hold the element
         
         if (isEmpty()) { // If the stack is empty, insert the new element at the end
             head = newNode;
             tail = newNode;
             size++;
         } else { // Otherwise, insert the new element before the last element
             newNode.next = head;
             head = newNode;
             size++;
         }
     }
     
     public int pop() {
         if (isEmpty()) return -1; // If the stack is empty, throw an error and return a negative value
         
         Node temp = head;
         int x = temp.value;
         
         if (!isFull()) { // If the stack is not full, remove the element from the head
             head = head.next;
             size--;
         }
         
         // Return popped element
         return x;
     }
     
     public boolean isEmpty() {
         return head == null;
     }
     
     public int top() {
         if (isEmpty()) return -1; // If the stack is empty, throw an error and return a negative value
         
         Node temp = head;
         return temp.value;
     }
     
     public void clearStack() {
         head = tail = null;
         size = 0;
     }
}
