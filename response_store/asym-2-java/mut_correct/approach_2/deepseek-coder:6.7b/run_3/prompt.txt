public final class AsymQ2 {
    // `a` has length M; `b` is sorted ascending and has length N.
    // Your analysis should be based on M and N.
    public static int f(int[] a, int[] b) {
        int m = a.length;
        int s = 0;
        for (int i = 0; i < m; i++) {
            int t = g(b, a[i]);
            if (t >= 0) s += t;
        }
        return s;
    }

    private static int g(int[] b, int x) {
        int l = 0;
        int r = b.length - 1;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (b[mid] == x) return mid;
            if (b[mid] < x) l = mid + 1;
            else r = mid - 1;
        }
        return -1;
    }
}
