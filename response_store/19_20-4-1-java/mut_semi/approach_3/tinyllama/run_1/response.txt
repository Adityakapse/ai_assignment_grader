2) "Tree Properties":
public boolean isBST(Node root) {
    return isBSTRecursive(root, root);
}

private boolean isBSTRecursive(Node root, Node prev) {
    if (prev == null || root.val < prev.val) { // Base case: root is root
        return true;
    } else if (isBSTRecursive(root.left, root)) { // Left child BST
        return true;
    } else if (isBSTRecursive(root.right, root)) { // Right child BST
        return true;
    } else { // Root is not a BST
        return false;
    }
}

3) "Search Tree":
public boolean search(Node root, int key) {
    if (root == null || root.val != key) { // Base case: root is null or not found
        return false;
    } else if (key < root.val) { // Recursive search left subtree
        return searchLeft(root);
    } else if (key > root.val) { // Recursive search right subtree
        return searchRight(root);
    } else { // Root is in tree
        return true;
    }
}

private boolean searchLeft(Node root) {
    if (root == null || root.left == null) {
        return false;
    } else {
        return searchLeft(root.left);
    }
}

public boolean search(Node root, int key) {
    if (root == null || root.val != key) { // Base case: root is null or not found
        return false;
    } else if (key < root.val) { // Recursive search left subtree
        return searchLeft(root);
    } else if (key > root.val) { // Recursive search right subtree
        return searchRight(root);
    } else { // Root is in tree
        return true;
    }
}

private boolean searchLeft(Node root) {
    Node curr = root; // Current node to search
    while (curr.left != null) {
        curr = curr.left;
    }
    return curr == null || curr.val < root.val;
}

5) "Search TRE":
public boolean search(Node root, int key) {
    if (root == null || root.val != key) { // Base case: root is null or not found
        return false;
    } else if (key < root.val) { // Recursive search left subtree
        return searchLeft(root);
    } else if (key > root.val) { // Recursive search right subtree
        return searchRight(root);
    } else { // Root is in tree
        return true;
    }
}

private boolean searchLeft(Node root) {
    Node curr = root, prev = null;
    while (curr != null && curr.left != root) {
        prev = curr;
        curr = curr.left;
    }
    if (curr == null || curr.val < root.val) { // Not found in left subtree
        return false;
    } else {
        prev.right = curr.right;
        return true;
    }
}
