    """Recursive function to sort an array in ascending order."""
    # base case: no elements in the list
    if n == 0:
        return arr
    
    # recursive case: divide into two halves
    left, right = divmod(arr[:n//2], n)
    
    # sort each half recursively
    left_sorted = sort_list(left, n)
    right_sorted = sort_list(right, n)
